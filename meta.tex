\newcommand{\III}[9]{%
  {}^{#1}_{#2} I^{#3}_{#4}
  \left(
    \begin{matrix}
      #5 \\ {}\\ #6
    \end{matrix}
    \,\middle|\,
    \begin{matrix}
      #7 \\ #8 \\ #9
    \end{matrix}
  \right)
}
\newcommand{\II}[7]{%
  {}^{#1}_{#2} I^{#3}_{#4}
  \left( #5, #6
    \,\middle|\,
    #7
  \right)
}
