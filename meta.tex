  {}^{#1}_{#2} I^{#3}_{#4}
  \left(
    \begin{matrix}
      #5 \\ {}\\ #6
    \end{matrix}
    \,\middle|\,
    \begin{matrix}
      #7 \\ #8 \\ #9
    \end{matrix}
  \right)
}

  \right)
